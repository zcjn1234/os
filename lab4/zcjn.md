void proc_run(struct proc_struct *proc) {     
  if (proc != current) {         
   bool intr_flag; 
     struct proc_struct *prev = current, *next = proc;         
local_intr_save(intr_flag);//关中断        
 { 
            current = proc;

 } 
        local_intr_restore(intr_flag);//开中断    
 } 
} 
 
首先关中断，然后准备保存现场，切换堆栈，lcr3指令用于切换页目录表基址，实际上next->cr3和当前的cr3是完全一样的，这里只是为了体现切换页目录这样一个过程。

switch_to:                      
# switch_to(from, to)    
 # save from's registers 
movl 4(%esp), %eax         
 # eax points to from 
popl 0(%eax)                # 保存返回地址到context->eip    
 movl %esp, 4(%eax)    
 movl %ebx, 8(%eax)

movl %edx, 16(%eax)
